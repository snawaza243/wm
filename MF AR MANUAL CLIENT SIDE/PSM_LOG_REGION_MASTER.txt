CREATE OR REPLACE PROCEDURE WEALTHMAKER.PSM_LOG_REGION_MASTER(
P_FOR        IN VARCHAR2,
P_BY         IN VARCHAR2,
P_BY_VALUE   IN VARCHAR2,
P_LOG_ID     IN VARCHAR2,
P_ROLE_ID    IN VARCHAR2,
P_CURSOR        OUT SYS_REFCURSOR

) AS
BEGIN

    IF P_FOR = 'MF_AR_MANUAL_RECO' THEN  
        IF P_BY IS NOT NULL THEN
            IF P_BY = 'CHANNEL' THEN
                OPEN P_CURSOR FOR
                    SELECT DISTINCT RM.REGION_NAME AS REGION_NAME, RM.REGION_ID AS REGION_CODE
                    FROM REGION_MASTER RM
                    JOIN BRANCH_MASTER BM ON RM.REGION_ID = BM.REGION_ID
                    WHERE 1=1
                    AND BM.BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID = P_LOG_ID AND ROLE_ID=P_ROLE_ID)  
                    AND BM.CATEGORY_ID NOT IN ('1004', '1005', '1006') 
                    AND BM.BRANCH_TAR_CAT = P_BY_VALUE
                    ORDER BY RM.REGION_NAME;

            END IF;
        ELSE
            -- BRANCH LIST FOR MF_AR_MANUAL_RECO  
            OPEN P_CURSOR FOR 
                    SELECT DISTINCT RM.REGION_NAME AS REGION_NAME, RM.REGION_ID AS REGION_CODE FROM REGION_MASTER RM
                    JOIN BRANCH_MASTER BM ON RM.REGION_ID = BM.REGION_ID
                    WHERE 1=1
                    AND BM.BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID = P_LOG_ID AND ROLE_ID=P_ROLE_ID)  
                    AND BM.CATEGORY_ID NOT IN ('1004', '1005', '1006') 
                    ORDER BY RM.REGION_NAME;
        END IF;
    ELSE
        OPEN P_CURSOR FOR 
            SELECT DISTINCT RM.REGION_NAME AS REGION_NAME, RM.REGION_ID AS REGION_CODE
            FROM REGION_MASTER RM
            JOIN BRANCH_MASTER BM ON RM.REGION_ID = BM.REGION_ID
            WHERE 1=1
            AND BM.BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID = P_LOG_ID AND ROLE_ID=P_ROLE_ID)  
            ORDER BY RM.REGION_NAME;
    END IF;
END;
