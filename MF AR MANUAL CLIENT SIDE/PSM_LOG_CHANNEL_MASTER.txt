CREATE OR REPLACE PROCEDURE WEALTHMAKER.PSM_LOG_CHANNEL_MASTER(
P_FOR        IN VARCHAR2,
P_BY         IN VARCHAR2,
P_BY_VALUE   IN VARCHAR2,
P_LOG_ID     IN VARCHAR2,
P_ROLE_ID    IN VARCHAR2,
P_CURSOR        OUT SYS_REFCURSOR

) IS 
BEGIN 

    IF P_FOR = 'MF_AR_MANUAL_RECO' THEN        
        OPEN P_CURSOR FOR 
            SELECT ITEMNAME, ITEMSERIALNUMBER FROM FIXEDITEM
            WHERE 1=1
            AND ITEMTYPE = '26' 
            ORDER BY ITEMNAME;
    ELSE
        OPEN P_CURSOR FOR 
        SELECT ITEMNAME, ITEMSERIALNUMBER FROM FIXEDITEM
        WHERE 1=1 
        --AND BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID = P_LOG_ID AND ROLE_ID=P_ROLE_ID) 
        ORDER BY ITEMNAME;
    END IF;
END ;
