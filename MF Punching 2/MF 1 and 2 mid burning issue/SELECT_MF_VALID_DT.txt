SELECT COMMON_ID,A.BUSI_RM_CODE,A.BUSI_BRANCH_CODE,A.BUSI_TR_DATE,A.SCH_CODE,A.INV_CODE,B.BRANCH_NAME,A.EXPENSE 
FROM TB_DOC_UPLOAD A,BRANCH_MASTER B WHERE A.BUSI_BRANCH_CODE=B.BRANCH_CODE 
AND  TRAN_TYPE='MF' AND VERIFICATION_FLAG='1' AND REJECTION_STATUS='0'  AND PUNCHING_FLAG='0'
AND DOC_UPLOAD_DATE>='01-JUN-2024';
