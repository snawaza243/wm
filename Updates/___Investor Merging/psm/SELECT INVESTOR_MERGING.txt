--INVESTOR SERACH CLIENT
SELECT CLIENT_NAME,CLIENT_CODE,A.ADDRESS1,A.ADDRESS2,C.CITY_NAME,B.BRANCH_NAME,A.PHONE,E.RM_NAME,KYC 
            FROM BRANCH_MASTER B,EMPLOYEE_MASTER E,CLIENT_MASTER A,CITY_MASTER C 
            WHERE E.SOURCE=B.BRANCH_CODE AND A.RM_CODE=E.RM_CODE AND A.CITY_ID=C.CITY_ID(+) AND CLIENT_NAME IS NOT NULL  
            AND BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID='49239' AND ROLE_ID='22')
            --AND TO_CHAR(A.CLIENT_CODE)= '45678903'
            AND UPPER(A.CLIENT_NAME) LIKE '%ARJUN%';


--INVESTOR SERACH agent
SELECT A.MOBILE, A.AGENT_NAME AS CLIENT_NAME, A.AGENT_CODE AS CLIENT_CODE,A.ADDRESS1,A.ADDRESS2,C.CITY_NAME,B.BRANCH_NAME,A.PHONE,E.RM_NAME,
(SELECT I.KYC FROM INVESTOR_MASTER I WHERE I.SOURCE_ID = A.AGENT_CODE AND ROWNUM = 1) AS KYC 
FROM BRANCH_MASTER B,EMPLOYEE_MASTER E,AGENT_MASTER A,CITY_MASTER C
WHERE E.SOURCE=B.BRANCH_CODE AND A.RM_CODE=E.RM_CODE AND A.CITY_ID=C.CITY_ID(+) AND A.AGENT_NAME IS NOT NULL 
AND BRANCH_CODE IN (SELECT BRANCH_ID FROM USERDETAILS_JI WHERE LOGIN_ID='38387' AND ROLE_ID='261') 
--AND TO_CHAR(A.AGENT_CODE)= '34567890' AND UPPER(A.AGENT_NAME) LIKE '%ARJUN%' 
AND UPPER(A.PHONE) LIKE '%30040944%'AND UPPER(A.MOBILE) LIKE  '%9879148943%'
