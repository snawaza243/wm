function onRowClick(row) {
    var grid = document.getElementById("<%= GridTransaction.ClientID %>");
    var checkboxes = grid.querySelectorAll("input[type='checkbox'].select-row-checkbox");
    var rows = grid.getElementsByTagName("tr");

    // Uncheck all checkboxes and reset row backgrounds
    for (var i = 0; i < checkboxes.length; i++) {
        checkboxes[i].checked = false;
        if (rows[i + 1]) { // skip header row
            rows[i + 1].style.backgroundColor = ""; // remove highlight
        }
    }

    // Check current row's checkbox
    var checkbox = row.querySelector("input[type='checkbox'].select-row-checkbox");
    if (checkbox) {
        checkbox.checked = true;
        row.style.backgroundColor = "#cce5ff"; // light blue highlight
        onRowSelect(checkbox); // your existing function
    }
}
